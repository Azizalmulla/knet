### 1. Update your Macchiato renderer:

```typescript
// lib/jsonresume/themes.ts
import macchiato from 'jsonresume-theme-macchiato'
import elegant from 'jsonresume-theme-elegant'
import stackoverflow from 'jsonresume-theme-stackoverflow'
import kendall from 'jsonresume-theme-kendall'

const themes = {
  macchiato,
  elegant,
  stackoverflow,
  kendall,
}

export async function renderWithTheme(
  cv: any, 
  themeName: string = 'elegant'
): Promise<string> {
  const theme = themes[themeName] || themes.macchiato
  const resume = mapToJsonResume(cv)
  
  return typeof theme === 'function' 
    ? theme(resume) 
    : theme.render(resume)
}
```

### 2. Update API route:

```typescript
// app/api/cv/macchiato/html/route.ts
export async function POST(request: Request) {
  const { cv, theme = 'elegant' } = await request.json()
  const html = await renderWithTheme(cv, theme)
  return new Response(html, {
    headers: { 'Content-Type': 'text/html' }
  })
}
```
